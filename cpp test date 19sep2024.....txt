#include <bits/stdc++.h>
using namespace std;

int main() 
{
    int n;
    cout << "Enter the number for printing partern : ";
    cin >> n;
    for (int i=1;i<=n;i++) 
    {
        cout<<"red ";
    }
    {        
        for (int j=1;j<=n;j++) 
        {       
            cout <<"blue ";
        }
        cout<<"\n";
    }
    return 0;
}
