#include <bits/stdc++.h>

int main() 
{
    int number, originalNumber, remaindernum, result = 0, numDigits = 0;

    cout << "Enter a number: ";
    cin >> number;
    originalNumber = number;

    int temp = number;
    while (temp != 0) 
    {
        temp /= 10;
        ++numDigits;
    }
    while (originalNumber != 0) 
    {
        remaindernum = originalNumber % 10;
        result+=pow(remaindernum,numDigits);
        originalNumber /= 10;
    }
    if (result==number) 
    {
        cout<<number<<" is a Armstrong number.\n";
    }
    else 
    {
        cout << number << " is not a Armstrong number.\n";
    }
    return 0;
}

q.53. Write a for loop to find the sum of the digits of each number in the Fibonacci series
up to 10 terms........


#include <bits/stdc++.h>
using namespace std;

int main() 
{
    int n = 10;
    int num1 = 0,num2 = 1,num3;
    cout << "Fibonacci Series up to " << n << " terms:" << "\n";
    for (int i = 0; i < n; ++i) 
    {
        if (i <= 1)
          num3 = i;
        else 
        {
            num3 = num1 + num2;
            num1 = num2;
            num2 = num3;
        }
        cout << num3 << " ";
        int sum = 0;
        for (int num=num3;num!=0;num/=10)
        {
            sum += num % 10;
        }
        cout << "Sum of digits: " << sum << "\n";
    }
    return 0;
}

q.55. Write a for loop to find the sum of all digits in a given range of numbers.....


#include <iostream>
using namespace std;

int main() 
{
    int sRange, eRange;
    cout << "Enter the start range: ";
    cin >> sRange;
    cout << "Enter the end range: ";
    cin >> eRange;
    int totalSum = 0;
    for (int num=sRange;num<=eRange;num++) 
    {
        int currentNumber=num;
        int sumOftheDigits=0;
        while(currentNumber!=0) 
        {
            sumOftheDigits+=currentNumber%10;
            currentNumber/=10;
        }
        totalSum+=sumOftheDigits;
    }
    cout<<"Sum of digits in the range from "<<sRange<<" to "<<eRange<<" is: "<<totalSum<<"\n";
    return 0;
}

Enter the end range: 20

{

    cout << "Enter a number: ";
    cin >> number;
    originalNumber = number;

    {
        reversedNumber=reversedNumber*10+temp%10;
