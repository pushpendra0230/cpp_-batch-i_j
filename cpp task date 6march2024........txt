#include <bits/stdc++.h>
using namespace std;
int main()
{
    int number,sum=0,last_digit;
    cout<<"enter the number :\n";
    cin>>number;
    int temp=number;
    while(number>0)
    {
        last_digit=number%10;
        sum=sum+last_digit*last_digit*last_digit;
        number=number/10;
    }
    if(temp==sum)
    {
        cout<<"this number is Armstrong";
    }
    else
    {
        cout<<"this number is Not a Armstrong";
    }
    return 0;
}
